## Описание 

``` bsl
 Создает экземпляр обработки для обхода коллекций.
 
 // Параметры:
//  Коллекция - ТаблицаЗначений, 
//              Массив Из Структура, 
//              Массив Из СтрокаТаблицыЗначений, 
//              ДанныеФормыКоллекция,
//              НаборЗаписей,
//              ТабличнаяЧасть - любая коллекция    
// элементов, доступная для обхода циклом Для каждого Из.
 	
// Возвращаемое значение:
// 	ОбработкаОбъект.ОбходКоллекции - обработка для обхода 
// коллекций.

 // Пример:
 
	Запрос = Новый Запрос;
	Запрос.Текст = 
		"ВЫБРАТЬ
		|	Данные.ПроизводственныйКалендарь КАК ПроизводственныйКалендарь,
		|	Данные.Год КАК Год,
		|	Данные.Месяц КАК Месяц,
		|	Данные.КалендарныхДней,
		|	Данные.РабочихДней
		|ИЗ
		|	РегистрСведений.ДанныеПроизводственногоКалендаряПомесячно КАК Данные
		|УПОРЯДОЧИТЬ ПО
		|	ПроизводственныйКалендарь,
		|	Год,
		|	Месяц";
	
	Обход = Обработки.ОбходКоллекции.НовыйОбход(Запрос.Выполнить().Выгрузить());
	
	Пока Обход.СледующийПоЗначениюПоля("ПроизводственныйКалендарь") Цикл
		Сообщить("Начало календаря " + Обход.ТекущиеДанные()["ПроизводственныйКалендарь"]);
		Пока Обход.СледующийПоЗначениюПоля("Год") Цикл
			Сообщить("Начало года " + Обход.ТекущиеДанные()["Год"]);
			Пока Обход.Следующий() Цикл
				Сообщить("Месяц " + Обход.ТекущиеДанные()["Месяц"]);
			КонецЦикла;
			Сообщить("Конец года " + Обход.ТекущиеДанные()["Год"]);
		КонецЦикла;
		Сообщить("Конец календаря " + Обход.ТекущиеДанные()["ПроизводственныйКалендарь"]);
	КонецЦикла;

```
